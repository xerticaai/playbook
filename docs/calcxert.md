        const EXCHANGE_RATE = 5.50;

        // =====================================
        // TAB: VENDAS (BDM & SS)
        // =====================================
        function calcVendas() {
            const quota = document.getElementById('vendas-quota').value;
            const netPlat = parseFloat(document.getElementById('vendas-plat').value) || 0;
            const netServ = parseFloat(document.getElementById('vendas-serv').value) || 0;
            const netSol = parseFloat(document.getElementById('vendas-sol').value) || 0;
            const hasSS = document.getElementById('vendas-hasSS').checked;

            const servSol = netServ + netSol;
            const totalNet = netPlat + servSol;
            
            document.getElementById('vendas-total-net').innerText = fmtUSD.format(totalNet);
            document.getElementById('vendas-servsol-net').innerText = fmtUSD.format(servSol);

            let ratePlat = 0, rateServSol = 0;
            let feedback = "";

            if (quota === '<80') {
                feedback = "<span class='text-xred'><i class='fa-solid fa-xmark'></i> Meta abaixo de 80%. Não gera comissão.</span>";
            } else if (quota === '80-100') {
                ratePlat = 0.05; rateServSol = 0.05;
                feedback = "Taxa Base de <b>5%</b> aplicada sobre o Booking.";
            } else {
                ratePlat = 0.05; rateServSol = 0.10;
                feedback = "<span class='text-xpink font-bold'><i class='fa-solid fa-rocket'></i> Acelerador Ouro! 10% (Serv/Sol) / 5% (Plat)</span>";
            }

            const poolPlat = netPlat * ratePlat;
            const poolServSol = servSol * rateServSol;
            const totalPool = poolPlat + poolServSol;

            document.getElementById('vendas-pool-usd').innerText = fmtUSD.format(totalPool);
            document.getElementById('vendas-pool-brl').innerHTML = `${fmtBRL.format(totalPool * EXCHANGE_RATE)} <span class="text-sm text-xgray font-normal">(Câmbio: 5,50)</span>`;
            document.getElementById('vendas-pool-exp').innerHTML = feedback;

            let bdmShare = 0, ssShare = 0;
            let ssBadge = "";

            if (hasSS && servSol >= 50000) {
                bdmShare = totalPool * 0.20;
                ssShare = totalPool * 0.80;
                document.getElementById('vendas-txt-bdm').innerText = "Dono da conta envolveu SS elegível (Fica com 20%).";
                document.getElementById('vendas-txt-ss').innerText = "SS tático fechou o negócio (Fica com 80%).";
                ssBadge = `<span class="tag tag-blue"><i class="fa-solid fa-check mr-1"></i> Elegível (>50k de Booking)</span>`;

        // =====================================
        // TAB: CUSTOMER SUCCESS (CS)
        // =====================================
        function calcCS() {
            const retention = document.getElementById('cs-retention').value || 0;
            const cob = document.getElementById('cs-cob').value || 0;
            const adop = document.getElementById('cs-adop').value || 0;
            const salary = document.getElementById('cs-salary').value || 0;
            const onTime = document.getElementById('cs-ontime').checked;

            const statusDiv = document.getElementById('cs-status-badge');
            let isBlocked = false;

            if (retention >= 95) {
                statusDiv.innerHTML = `<span class="tag tag-green text-sm px-4 py-2"><i class="fa-solid fa-unlock mr-2"></i> Bónus Destravado</span>`;
            } else {
                statusDiv.innerHTML = `<span class="tag tag-red text-sm px-4 py-2"><i class="fa-solid fa-lock mr-2"></i> Bloqueado (Retenção < 95%)</span>`;
                isBlocked = true;
            }

            // Calculo Bonus
            let bonusValue = 0;
            let finalPercent = 0;

            if (!isBlocked) {
                const nrrAttainment = Math.min((retention / 95) * 100, 150);
                const pondNRR = nrrAttainment * 0.40;
                const pondCob = cob * 0.25;
                const pondAdop = adop * 0.25;
                const pondCross = 100 * 0.10; // Hardcoded 100% cross-sell for simulation ease
                
                let basePercent = pondNRR + pondCob + pondAdop + pondCross;
                let multiplier = onTime ? 1.5 : 1.0;
                finalPercent = basePercent * multiplier;
                
                bonusValue = salary * (finalPercent / 100);
            }

            document.getElementById('cs-res-bonus-usd').innerText = fmtUSD.format(bonusValue);
            document.getElementById('cs-res-percent').innerText = `${finalPercent.toFixed(1)}% do Salário Base (Trimestral)`;

            // Calculo Upsell
            const upPlat = parseFloat(document.getElementById('cs-up-plat').value) || 0;
            const upServ = parseFloat(document.getElementById('cs-up-serv').value) || 0;
            
            const upPool = (upPlat + upServ) * 0.05; // Assuming flat 5%
            const upComission = upPool * 0.20; // 20% engagement

            const upWarningDiv = document.getElementById('cs-upsell-warning');
            const resComUsd = document.getElementById('cs-res-com-usd');
            const resComBrl = document.getElementById('cs-res-com-brl');

            if (isBlocked && upComission > 0) {
                resComUsd.innerHTML = `<span class="text-xgray text-lg line-through mr-2">${fmtUSD.format(upComission)}</span> <span class="text-xred">$0.00</span>`;
                resComBrl.innerText = "R$ 0,00";
                upWarningDiv.innerHTML = `<span class="text-xred text-xs font-bold"><i class="fa-solid fa-triangle-exclamation mr-1"></i> Retenção baixa anulou a sua comissão de vendas!</span>`;
            } else {
                resComUsd.innerText = fmtUSD.format(upComission);
                resComBrl.innerText = fmtBRL.format(upComission * EXCHANGE_RATE);
                if (upComission > 0) {
                    upWarningDiv.innerHTML = `<span class="text-xgreen text-xs font-bold"><i class="fa-solid fa-check mr-1"></i> Valor libertado para pagamento (Regra aplicável).</span>`;
                } else {
                    upWarningDiv.innerHTML = "";
                }
            }
        }

            const attachOk = document.getElementById('ce-attach').checked;

                if (attachOk) validNet += plat;
